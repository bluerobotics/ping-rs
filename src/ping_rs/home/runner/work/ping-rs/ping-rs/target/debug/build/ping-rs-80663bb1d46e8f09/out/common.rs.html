use crate::serialize::PingMessage;
use crate::serialize::Serialize;
use bytes::BufMut;
#[cfg(feature = "serde")]
use serde::{Deserialize, Serialize};
#[derive(Debug, Clone, PartialEq, Default)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
pub struct PingProtocolHead {
    pub source_device_id: u8,
    pub destiny_device_id: u8,
}
#[derive(Debug, Clone, PartialEq)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
pub enum Messages {
    ProtocolVersion(ProtocolVersionStruct),
    Ack(AckStruct),
    Nack(NackStruct),
    GeneralRequest(GeneralRequestStruct),
    AsciiText(AsciiTextStruct),
    SetDeviceId(SetDeviceIdStruct),
    DeviceInformation(DeviceInformationStruct),
}
#[derive(Debug, Clone, PartialEq, Default)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
#[doc = "The protocol version"]
pub struct ProtocolVersionStruct {
    #[doc = "Information about message destiny and source"]
    pub header: PingProtocolHead,
    #[doc = "Protocol version major number."]
    pub version_major: u8,
    #[doc = "Protocol version minor number."]
    pub version_minor: u8,
    #[doc = "Protocol version patch number."]
    pub version_patch: u8,
    #[doc = "reserved"]
    pub reserved: u8,
}
impl Serialize for ProtocolVersionStruct {
    fn serialize(self, buffer: &mut [u8]) -> usize {
        buffer[0usize..1usize].copy_from_slice(&self.version_major.to_le_bytes());
        buffer[1usize..2usize].copy_from_slice(&self.version_minor.to_le_bytes());
        buffer[2usize..3usize].copy_from_slice(&self.version_patch.to_le_bytes());
        buffer[3usize..4usize].copy_from_slice(&self.reserved.to_le_bytes());
        4usize
    }
}
#[derive(Debug, Clone, PartialEq, Default)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
#[doc = "Acknowledged."]
pub struct AckStruct {
    #[doc = "Information about message destiny and source"]
    pub header: PingProtocolHead,
    #[doc = "The message ID that is ACKnowledged."]
    pub acked_id: u16,
}
impl Serialize for AckStruct {
    fn serialize(self, buffer: &mut [u8]) -> usize {
        buffer[0usize..2usize].copy_from_slice(&self.acked_id.to_le_bytes());
        2usize
    }
}
#[derive(Debug, Clone, PartialEq, Default)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
#[doc = "Not acknowledged."]
pub struct NackStruct {
    #[doc = "Information about message destiny and source"]
    pub header: PingProtocolHead,
    #[doc = "The message ID that is Not ACKnowledged."]
    pub nacked_id: u16,
    #[doc = "ASCII text message indicating NACK condition. (not necessarily NULL terminated)"]
    nack_message: String,
}
impl Serialize for NackStruct {
    fn serialize(self, buffer: &mut [u8]) -> usize {
        buffer[0usize..2usize].copy_from_slice(&self.nacked_id.to_le_bytes());
        let string_size = self.nack_message.len();
        let final_size = 2usize + string_size;
        buffer[2usize..final_size].copy_from_slice(self.nack_message.as_bytes());
        buffer[final_size] = 0;
        final_size + 1usize
    }
}

#[derive(Debug, Clone, PartialEq, Default)]
#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
#[doc = "Device information"]
pub struct DeviceInformationStruct {
    #[doc = "Information about message destiny and source"]
    pub header: PingProtocolHead,
    #[doc = "Device type. 0: Unknown; 1: Ping Echosounder; 2: Ping360"]
    pub device_type: u8,
    #[doc = "device-specific hardware revision"]
    pub device_revision: u8,
    #[doc = "Firmware version major number."]
    pub firmware_version_major: u8,
    #[doc = "Firmware version minor number."]
    pub firmware_version_minor: u8,
    #[doc = "Firmware version patch number."]
    pub firmware_version_patch: u8,
    #[doc = "reserved"]
    pub reserved: u8,
}
impl Serialize for DeviceInformationStruct {
    fn serialize(self, buffer: &mut [u8]) -> usize {
        buffer[0usize..1usize].copy_from_slice(&self.device_type.to_le_bytes());
        buffer[1usize..2usize].copy_from_slice(&self.device_revision.to_le_bytes());
        buffer[2usize..3usize].copy_from_slice(&self.firmware_version_major.to_le_bytes());
        buffer[3usize..4usize].copy_from_slice(&self.firmware_version_minor.to_le_bytes());
        buffer[4usize..5usize].copy_from_slice(&self.firmware_version_patch.to_le_bytes());
        buffer[5usize..6usize].copy_from_slice(&self.reserved.to_le_bytes());
        6usize
    }
}
impl PingMessage for Messages {
    fn message_name(&self) -> &'static str {
        match self {
            Messages::ProtocolVersion(..) => "protocol_version",
            Messages::Ack(..) => "ack",
            Messages::Nack(..) => "nack",
            Messages::GeneralRequest(..) => "general_request",
            Messages::AsciiText(..) => "ascii_text",
            Messages::SetDeviceId(..) => "set_device_id",
            Messages::DeviceInformation(..) => "device_information",
        }
    }
    fn message_id(&self) -> u16 {
        match self {
            Messages::ProtocolVersion(..) => 5u16,
            Messages::Ack(..) => 1u16,
            Messages::Nack(..) => 2u16,
            Messages::GeneralRequest(..) => 6u16,
            Messages::AsciiText(..) => 3u16,
            Messages::SetDeviceId(..) => 100u16,
            Messages::DeviceInformation(..) => 4u16,
        }
    }
    fn message_id_from_name(name: &str) -> Result<u16, &'static str> {
        match name {
            "protocol_version" => Ok(5u16),
            "ack" => Ok(1u16),
            "nack" => Ok(2u16),
            "general_request" => Ok(6u16),
            "ascii_text" => Ok(3u16),
            "set_device_id" => Ok(100u16),
            "device_information" => Ok(4u16),
            _ => Err("Invalid message name."),
        }
    }
    fn serialize(self, buffer: &mut [u8]) -> usize {
        match self {
            Messages::ProtocolVersion(content) => content.serialize(buffer),
            Messages::Ack(content) => content.serialize(buffer),
            Messages::Nack(content) => content.serialize(buffer),
            Messages::GeneralRequest(content) => content.serialize(buffer),
            Messages::AsciiText(content) => content.serialize(buffer),
            Messages::SetDeviceId(content) => content.serialize(buffer),
            Messages::DeviceInformation(content) => content.serialize(buffer),
        }
    }
}
